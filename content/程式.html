function setDisplayDigit(number, isgreen)
    for i = 0, 8, 1 do
        if (number == i) then
            for j = 0, 6, 1 do
                local ng = sim.getObject('./nG_['..j..']')
                local n = sim.getObject('./N_['..j..']')

                    sim.setShapeColor(n, nil, sim.colorcomponent_ambient_diffuse, {0, 1, 0})
                    sim.setShapeColor(_g_, nil, sim.colorcomponent_ambient_diffuse, {0, 1, 0})

                    end
                else    
                sim.setShapeColor(ng, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 1})
                end
            end
        end
