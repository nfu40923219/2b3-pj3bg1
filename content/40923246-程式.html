轉盤計分

function sysCall_init()
local math = require("math")
joint1= sim.getObject('./joint1')
joint2= sim.getObject('./joint2')
ball= sim.getObject('./ball')
sensor = sim.getObject('./sensor')
r1 = 360
r2 = 360
r3 = 360
r4 = 360
sim.setJointTargetPosition(joint1, math.rad(r1+180))
sim.setJointTargetPosition(joint2, math.rad(r2+180))

initialballPosition = sim.getObjectPosition(ball, -1)
initialballOrientation = sim.getObjectOrientation(ball, -1)
end
function sysCall_actuation()
result = sim.readProximitySensor(sensor)
if(r2>0 and result>0)then
sim.setObjectPosition(ball, -1, initialballPosition)
sim.setObjectOrientation(ball, -1, initialballOrientation)
r1=r1-36
sim.setJointTargetPosition(joint1, math.rad(r1+180))
if(r1==0) then
r1=360
r2=r2-36
sim.setJointTargetPosition(joint1, math.rad(r1+180))
sim.setJointTargetPosition(joint2, math.rad(r2+180))
end
end
end

    s5={1,0,0,1,1,1,1}

    for j = 0,6,1 do

    end
    for j = 0,6,1 do
        local n = sim.getObject('./N_['..j..']')
            sim.setShapeColor(n, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 0})
    end
    for go = 0,14,1 do
        local goal = sim.getObject('./goal_['..go..']')
            sim.setShapeColor(goal, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 1})
    end
    _g_=sim.getObject('./_nG_')
        sim.setShapeColor(_g_, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 0})
    for k = 1, 8 do
        bubbleRobHandles[k] = sim.getObjectHandle('./bubbleRob'..k)
    end
end
function sysCall_actuation()
    result = sim.readProximitySensor(sensor)
    for k = 1,8,1 do
        local Handle = sim.checkCollision(bubbleRobHandles[k], ballHandle)
        if Handle ~= 0 then
            number = k
            setDisplayDigit(k, k <= 4)
            break
        end
    end
    if result>0 then
        count2=100
        for go = 0,14,1 do
            local goal = sim.getObject('./goal_['..go..']')
            sim.setShapeColor(goal, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 0})
        end
    elseif (result==0 and count2 >0) then
        count2=count2-1
        if count2 ==0 then
            for go = 0,14,1 do
                local goal = sim.getObject('./goal_['..go..']')
                sim.setShapeColor(goal, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 1})
            end
            for i = 0, 8, 1 do
                if (number == i) then
                    for j = 0, 6, 1 do
                        local ng = sim.getObject('./nG_['..j..']')
                        local n = sim.getObject('./N_['..j..']')
                        sim.setShapeColor(n, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 0})
                        sim.setShapeColor(_g_, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 0})
                        if (s[i+1][j+1] == 1) then
                            sim.setShapeColor(ng, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 0})
                        end
                    end
                end
            end
        end
    end
end
function setDisplayDigit(number, isgreen)
    for i = 0, 8, 1 do
        if (number == i) then
            for j = 0, 6, 1 do
                local ng = sim.getObject('./nG_['..j..']')
                local n = sim.getObject('./N_['..j..']')
                if isgreen then
                    sim.setShapeColor(n, nil, sim.colorcomponent_ambient_diffuse, {0, 1, 0})
                    sim.setShapeColor(_g_, nil, sim.colorcomponent_ambient_diffuse, {0, 1, 0})
                else
                    sim.setShapeColor(n, nil, sim.colorcomponent_ambient_diffuse, {1, 0, 0})
                    sim.setShapeColor(_g_, nil, sim.colorcomponent_ambient_diffuse, {1, 0, 0})
                end
                if (s[i+1][j+1] == 1) then
                    if isgreen then
                    sim.setShapeColor(ng, nil, sim.colorcomponent_ambient_diffuse, {0, 1, 0})
                    else
                    sim.setShapeColor(ng, nil, sim.colorcomponent_ambient_diffuse, {1, 0, 0})
                    end
                else    
                sim.setShapeColor(ng, nil, sim.colorcomponent_ambient_diffuse, {1, 1, 1})
                end
            end
        end
    end
end
